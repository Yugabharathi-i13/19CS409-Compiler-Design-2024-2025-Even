anb.y
```
%{
#include <stdio.h>
#include <stdlib.h>

extern int b_count;  // The count of 'b's, declared in the Lex file

void yyerror(char *s) {
    fprintf(stderr, "Error: %s\n", s);
}

int yylex();  // The lexer function generated by Lex
%}

%token A B

%%

start:
    A b_sequence { 
        if (b_count >= 10) 
            printf("Valid string: matches anb where n >= 10\n");
        else 
            printf("Invalid: number of 'b's is less than 10\n");
    }
    ;

b_sequence:
    B b_sequence { /* Continue matching 'b's */ }
    | B { /* One 'b' is enough for now */ }
    ;

%%

int main() {
    printf("Enter a string: ");
    yyparse();  // Start parsing
    return 0;
}
```
